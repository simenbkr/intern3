Legg til Debian-kilder for PHP-pakker

sudo echo 'deb http://packages.dotdeb.org jessie all' >> /etc/apt/sources.list
sudo echo 'deb-src http://packages.dotdeb.org jessie all' >> /etc/apt/sources.list

sudo apt update

sudo apt install git curl mysql-server php7.0 libapache2-mod-php7.0 php7.0-cli php7.0-common php7.0-mbstring php7.0-gd php7.0-intl php7.0-xml php7.0-mysql php7.0-mcrypt php7.0-zip

Kjør mysql_secure_installation for å ordne sikker installasjon:
mysql_secure_installation

cd /var/www/
git clone bruker@dev.singsaker.no:/var/git/intern3.git
cd intern3
curl -sS https://getcomposer.org/installer | php
php composer.phar install

vim config.php
